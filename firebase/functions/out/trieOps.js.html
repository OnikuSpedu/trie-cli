/**
 * Recursively returns a copy of the trie with the string provided added to it
 * @param {string} str
 * @param {Trie} map
 * @return {Trie}
 */
function add(str, map) {
    if (str.length == 0) {
        map["self"] = true;
        return map;
    }
    else {
        if (map[str.substring(0, 1)]) {
            map[str.substring(0, 1)] = add(str.substring(1), map[str.substring(0, 1)]);
            return map;
        }
        else {
            map[str.substring(0, 1)] = add(str.substring(1), {});
            return map;
        }
    }
}
exports.add = add;

/**
 * Recursively created a trie without the specified string
 * @param {string} str
 * @param {Trie} map
 * @return {boolean}
 */
function remove(str, map) {
    if (str.length == 0) {
        if (map["self"]) {
            delete map["self"];
            return map;
        }
        return map;
    }
    else {
        if (map[str.substring(0, 1)]) {
            const next = remove(str.substring(1), map[str.substring(0, 1)]);
            if (Object.keys(next).length == 0) {
                delete map[str.substring(0, 1)];
            }
            else {
                map[str.substring(0, 1)] = next;
            }
        }
        return map;
    }
}

/**
 * Display helper function that recursively adds strings to words array
 *  @param {string} str
 *  @param {Trie} map
 *  @param {string[]} words
 */
function displayHelper(str, map, words) {
    if (map["self"]) {
        words.push(str);
    }
    Object.keys(map).forEach((key) => {
        if (key !== "self") {
            displayHelper(str + key, map[key], words);
        }
    });
}
/**
 * Returns words in a trie that start with the specified string
 *  @param {string} str
 *  @param {Trie} map
 *  @return {string[]}
 */
function autocomplete(str, map) {
    let newMap = Object.assign({}, map);
    for (let i = 0; i < str.length; i++) {
        newMap = newMap[str.charAt(i)];
    }
    let words = []; // eslint-disable-line
    displayHelper(str, newMap, words);
    return words;
}
exports.autocomplete = autocomplete;
